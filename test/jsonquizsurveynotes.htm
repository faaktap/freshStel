DROP TABLE IF EXISTS dkhs_survey;

CREATE TABLE IF NOT EXISTS dkhs_survey (
  surveyid int(11) NOT NULL AUTO_INCREMENT,
  ownerid int(11) NULL,
  createdate timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  quizdata varchar(2048) NOT NULL,
  PRIMARY KEY (surveyid)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

SHOW CREATE TABLE dkhs_quiz;

DROP TABLE IF EXISTS dkhs_quiz;

CREATE TABLE dkhs_quiz (
 quizid int(11) NOT NULL AUTO_INCREMENT,
 surveyid int(11) NOT NULL,
 ownerid int(11) NULL,
 createdate timestamp NOT NULL DEFAULT current_timestamp(),
 station varchar(40) NULL,
 quizkey varchar(20) NULL,
 quizdata varchar(2048) NOT NULL,
 PRIMARY KEY (quizid)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
